双指针 推荐

````java
class Solution {
    public String reverseWords(String s) {
        s = s.trim(); 
        int j = s.length() - 1, i = j;
        StringBuilder res = new StringBuilder();
        while(i >= 0) {
            while(i >= 0 && s.charAt(i) != ' ') 
            	i--;//这里控制跳出循环
            res.append(s.substring(i + 1, j + 1) + " ");
            // 跳过单词间空格
            while(i >= 0 && s.charAt(i) == ' ') 
            	i--; 
            // j指向下个单词的尾字符
        	j = i; 
        }
        return res.toString().trim(); // 转化为字符串并返回
    }
}
````
