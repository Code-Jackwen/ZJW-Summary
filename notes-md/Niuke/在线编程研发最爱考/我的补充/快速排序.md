快速排序

````java
public class Sort {
	public static void quickSort(int[] arr, int l, int h) {
        if (l > h)  return;
        int i = l, j = h, tag = arr[l];
        while (i < j) {
            while (tag <= arr[j] && i < j)  j--;
            while (tag >= arr[i] && i < j)  i++;
            if (i < j)   swap(arr,i,j);
        }
        swap(arr,i,l);
        quickSort(arr, l, j - 1);
        quickSort(arr, j + 1, h);
    }
    public static void swap(int[] arr,int a,int b){
        int t = arr[a];
        arr[a] = arr[b];
        arr[b] = t;
    }
}
````

非递归版本

````java
class Solution {
    public int[] sortArray(int[] nums) {
        int l = 0;
        int r = nums.length - 1;
        Stack<Integer> stack = new Stack<>();
        stack.push(r);
        stack.push(l);
        int i = 0;
        while (!stack.isEmpty()) {
            l = stack.pop();
            r = stack.pop();
            if (l < r) {
                i = quickSort(nums, l, r);
            }
            if (l < i - 1) {
                stack.push(i - 1);
                stack.push(l);
            }
            if (r > i + 1) {
                stack.push(r);
                stack.push(i + 1);
            }
        }
        return nums;
    }
    public static int quickSort(int[] arr, int l, int h) {
        int i = l, j = h, tag = arr[l];
        while (i < j) {
            while (tag <= arr[j] && i < j) j--;
            while (tag >= arr[i] && i < j) i++;
            if (i < j) swap(arr, i, j);
        }
        swap(arr, i, l);
        return i;
    }
    public static void swap(int[] arr, int a, int b) {
        int t = arr[a];
        arr[a] = arr[b];
        arr[b] = t;
    }
}
````
